  \node[state]         (3) [below right of=1] {$(s_3,q_0)$};

  \path (4) edge[red,line width=3pt]              node  {}(2)
            edge     [bend right,right]         node  {}(5)
               edge   [loop right]           node  {}(4);

  \path (2) edge              node  {}(init)
               edge[red,line width=3pt]             node  {}(3);

  \path (5)    edge  [bend right,right]           node  {}(4);
